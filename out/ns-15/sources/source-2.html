272     public double getDistance(LocationBean loc1, LocationBean loc2) {
273         double lat1 = Math.toRadians(loc1.latitude);
274         double lon1 = Math.toRadians(loc1.longitude);
275         double lat2 = Math.toRadians(loc2.latitude);
276         double lon2 = Math.toRadians(loc2.longitude);
277 
278         double angle = Math.acos(Math.sin(lat1) * Math.sin(lat2)
279                 + Math.cos(lat1) * Math.cos(lat2) * Math.cos(lon1 - lon2));
280 
281         double nauticalMiles = 60 * Math.toDegrees(angle);
282         double statuteMiles = STATUTE_MILES_PER_NAUTICAL_MILE * nauticalMiles;
283         return statuteMiles;
284     }
